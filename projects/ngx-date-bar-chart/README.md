## Installation
```
npm install ngx-date-bar-chart
```

## Usage
In your module add NgxDateBarChartModule to the imports:
```
@NgModule({
  declarations: [
    AppComponent
  ],
  imports: [
    BrowserModule,
    NgxDateBarChartModule
  ],
  providers: [],
  bootstrap: [AppComponent]
})
```
